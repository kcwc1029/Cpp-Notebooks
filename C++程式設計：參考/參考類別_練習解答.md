## 題目 4【多值回傳】

```cpp
#include <iostream>
using namespace std;

void calcStats(const int arr[], int size,
               int& sum, double& avg, int& maxVal, int& minVal) {
    sum = arr[0];
    maxVal = minVal = arr[0];

    for (int i = 1; i < size; i++) {
        sum += arr[i];
        if (arr[i] > maxVal) maxVal = arr[i];
        if (arr[i] < minVal) minVal = arr[i];
    }

    avg = (double)sum / size;
}

int main() {
    int arr[] = {5, 3, 8, 1, 9, 2, 7, 4, 6};
    int  n    = sizeof(arr) / sizeof(arr[0]);
    int  sum, maxVal, minVal;
    double avg;

    calcStats(arr, n, sum, avg, maxVal, minVal);

    cout << "總和：" << sum    << endl;
    cout << "平均：" << avg    << endl;
    cout << "最大：" << maxVal << endl;
    cout << "最小：" << minVal << endl;

    return 0;
}
```

**執行結果：**
```
總和：45
平均：5
最大：9
最小：1
```

## 題目 5【就地字串處理】

```cpp
#include <iostream>
#include <string>
using namespace std;

void reverseStr(string& s) {
    int left = 0, right = (int)s.length() - 1;
    while (left < right) {
        char temp = s[left];
        s[left]   = s[right];
        s[right]  = temp;
        left++;
        right--;
    }
}

void removeSpaces(string& s) {
    string result = "";
    for (char c : s) {
        if (c != ' ') result += c;
    }
    s = result;
}

int main() {
    string str = "Hello World !";
    cout << "原始：\"" << str << "\"" << endl;

    reverseStr(str);
    cout << "反轉後：\"" << str << "\"" << endl;

    removeSpaces(str);
    cout << "移除空白後：\"" << str << "\"" << endl;

    return 0;
}
```

**執行結果：**
```
原始："Hello World !"
反轉後："! dlroW olleH"
移除空白後："!dlroWolleH"
```

## 題目 6【泡沫排序 + 傳參考交換】

```cpp
#include <iostream>
using namespace std;

void mySwap(int& a, int& b) {
    int temp = a;
    a = b;
    b = temp;
}

void printArray(const int arr[], int size) {
    for (int i = 0; i < size; i++) {
        cout << arr[i];
        if (i < size - 1) cout << " ";
    }
    cout << endl;
}

void bubbleSort(int arr[], int size) {
    for (int i = 0; i < size - 1; i++) {
        cout << "第 " << i + 1 << " 輪：";
        for (int j = 0; j < size - i - 1; j++) {
            if (arr[j] > arr[j + 1]) {
                mySwap(arr[j], arr[j + 1]);
            }
        }
        printArray(arr, size);
    }
}

int main() {
    int arr[] = {64, 34, 25, 12, 22, 11, 90};
    int n = sizeof(arr) / sizeof(arr[0]);

    cout << "排序前：";
    printArray(arr, n);
    cout << endl;

    bubbleSort(arr, n);

    cout << "\n排序後：";
    printArray(arr, n);

    return 0;
}
```

**執行結果：**
```
排序前：64 34 25 12 22 11 90
第 1 輪：34 25 12 22 11 64 90
第 2 輪：25 12 22 11 34 64 90
第 3 輪：12 22 11 25 34 64 90
第 4 輪：12 11 22 25 34 64 90
第 5 輪：11 12 22 25 34 64 90
第 6 輪：11 12 22 25 34 64 90
排序後：11 12 22 25 34 64 90
```

## 題目 7【Range-for + 參考應用】

```cpp
#include <iostream>
using namespace std;

int main() {
    int arr[] = {-3, 7, -1, 5, -8, 2, -4, 6};
    int n = sizeof(arr) / sizeof(arr[0]);

    cout << "原始陣列：";
    for (int x : arr) cout << x << " ";
    cout << endl;

    // 1. 將負數改為正數（需要用 & 才能修改）
    for (int& x : arr) {
        if (x < 0) x = -x;
    }

    cout << "取絕對值後：";
    for (int x : arr) cout << x << " ";
    cout << endl;

    // 2. 計算總和（const auto& 只讀）
    int sum = 0;
    for (const auto& x : arr) {
        sum += x;
    }
    cout << "總和：" << sum << endl;

    // 3. 找最大最小值
    int maxVal = arr[0], minVal = arr[0];
    for (const auto& x : arr) {
        if (x > maxVal) maxVal = x;
        if (x < minVal) minVal = x;
    }
    cout << "最大值：" << maxVal << endl;
    cout << "最小值：" << minVal << endl;

    return 0;
}
```

**執行結果：**
```
原始陣列：-3 7 -1 5 -8 2 -4 6
取絕對值後：3 7 1 5 8 2 4 6
總和：36
最大值：8
最小值：1
```

## 題目 8【結構體與參考】

```cpp
#include <iostream>
#include <cmath>
using namespace std;

struct Point { double x, y; };

void move(Point& p, double dx, double dy) {
    p.x += dx;
    p.y += dy;
}

double distance(const Point& p1, const Point& p2) {
    double dx = p1.x - p2.x;
    double dy = p1.y - p2.y;
    return sqrt(dx * dx + dy * dy);
}

Point& closer(Point& p1, Point& p2, const Point& origin) {
    double d1 = distance(p1, origin);
    double d2 = distance(p2, origin);
    return d1 <= d2 ? p1 : p2;
}

void printPoint(const string& label, const Point& p) {
    cout << label << " = (" << p.x << ", " << p.y << ")" << endl;
}

int main() {
    Point A = {3, 4};
    Point B = {1, 1};
    Point origin = {0, 0};

    printPoint("A", A);
    printPoint("B", B);

    cout << "A 到 B 的距離：" << distance(A, B) << endl;
    cout << "A 到原點距離：" << distance(A, origin) << endl;
    cout << "B 到原點距離：" << distance(B, origin) << endl;

    // 找出較近的點並修改
    Point& near = closer(A, B, origin);
    cout << "\n離原點較近的是：(" << near.x << ", " << near.y << ")" << endl;

    near.x = 0;
    near.y = 0;
    cout << "將其移到原點後：" << endl;
    printPoint("A", A);
    printPoint("B", B);

    // 移動 A
    move(A, 2, -1);
    cout << "\n移動 A(+2,-1) 後：";
    printPoint("A", A);

    return 0;
}
```

**執行結果：**
```
A = (3, 4)
B = (1, 1)
A 到 B 的距離：3.60555
A 到原點距離：5
B 到原點距離：1.41421
離原點較近的是：(1, 1)
將其移到原點後：
A = (3, 4)
B = (0, 0)
移動 A(+2,-1) 後：A = (5, 3)
```

## 題目 9【自訂安全陣列】

```cpp
#include <iostream>
using namespace std;

class SafeArray {
    int data[100];
    int sz;

public:
    SafeArray(int n) : sz(n) {
        for (int i = 0; i < sz; i++) data[i] = 0;
    }

    int& at(int index) {
        if (index < 0 || index >= sz) {
            cout << "❌ 錯誤：索引 " << index
                 << " 超出範圍 [0, " << sz - 1 << "]，回傳 data[0]" << endl;
            return data[0];    // 回傳第一個元素的參考（不崩潰）
        }
        return data[index];
    }

    int size() const { return sz; }

    void print() const {
        cout << "[";
        for (int i = 0; i < sz; i++) {
            cout << data[i];
            if (i < sz - 1) cout << ", ";
        }
        cout << "]" << endl;
    }
};

int main() {
    SafeArray arr(5);

    cout << "初始狀態：";
    arr.print();

    // 正常存取與修改
    arr.at(0) = 10;
    arr.at(2) = 30;
    arr.at(4) = 50;
    cout << "修改後：";
    arr.print();

    // 讀取
    cout << "arr.at(2) = " << arr.at(2) << endl;

    // 越界存取（不崩潰，回傳 data[0] 的參考）
    cout << "\n測試越界：" << endl;
    arr.at(-1) = 99;    // 越界 → 修改到 data[0]
    arr.at(10);         // 越界 → 只印錯誤
    cout << "越界後陣列：";
    arr.print();

    return 0;
}
```

**執行結果：**
```
初始狀態：[0, 0, 0, 0, 0]
修改後：[10, 0, 30, 0, 50]
arr.at(2) = 30
測試越界：
❌ 錯誤：索引 -1 超出範圍 [0, 4]，回傳 data[0]
❌ 錯誤：索引 10 超出範圍 [0, 4]，回傳 data[0]
越界後陣列：[99, 0, 30, 0, 50]
```

## 題目 10【CSV 字串解析器】

```cpp
#include <iostream>
#include <string>
using namespace std;

bool parseCSV(const string& line,
              string& name, int& score, string& grade) {
    // 找第一個逗號
    int pos1 = line.find(',');
    if (pos1 == (int)string::npos) return false;

    // 找第二個逗號
    int pos2 = line.find(',', pos1 + 1);
    if (pos2 == (int)string::npos) return false;

    name  = line.substr(0, pos1);
    string scoreStr = line.substr(pos1 + 1, pos2 - pos1 - 1);
    grade = line.substr(pos2 + 1);

    // 嘗試轉換成績
    try {
        score = stoi(scoreStr);
    } catch (...) {
        return false;
    }

    // 基本驗證
    if (name.empty() || grade.empty()) return false;
    if (score < 0 || score > 100) return false;

    return true;
}

int main() {
    string testCases[] = {
        "Alice,92,A",
        "Bob,78,C",
        "Charlie,55,F",
        "invalid",
        "NoComma",
        "OnlyOne,comma"
    };

    cout << "=== 測試資料 ===" << endl;
    for (const string& line : testCases) {
        string name, grade;
        int score;

        if (parseCSV(line, name, score, grade)) {
            cout << "✅ \"" << line << "\""
                 << " → 姓名=" << name
                 << ", 分數=" << score
                 << ", 等第=" << grade << endl;
        } else {
            cout << "❌ \"" << line << "\" → 解析失敗" << endl;
        }
    }

    cout << "\n=== 互動模式（輸入空行結束）===" << endl;
    string line;
    getline(cin, line);    // 清除緩衝區
    while (true) {
        cout << "輸入 CSV（或空行結束）：";
        getline(cin, line);
        if (line.empty()) break;

        string name, grade;
        int score;
        if (parseCSV(line, name, score, grade)) {
            cout << "  → 姓名=" << name << ", 分數=" << score
                 << ", 等第=" << grade << endl;
        } else {
            cout << "  → 格式錯誤！請輸入「姓名,分數,等第」" << endl;
        }
    }

    cout << "程式結束。" << endl;
    return 0;
}
```

```
=== 測試資料 ===
✅ "Alice,92,A" → 姓名=Alice, 分數=92, 等第=A
✅ "Bob,78,C" → 姓名=Bob, 分數=78, 等第=C
✅ "Charlie,55,F" → 姓名=Charlie, 分數=55, 等第=F
❌ "invalid" → 解析失敗
❌ "NoComma" → 解析失敗
❌ "OnlyOne,comma" → 解析失敗
```
