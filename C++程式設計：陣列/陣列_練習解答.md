## 練習 7-2 解答：陣列旋轉

📄 **解答檔案：** `hw7_2_rotate_array.cpp`

```cpp
#include <iostream>
using namespace std;

// 反轉 arr[left..right]
void reverse(int arr[], int left, int right) {
    while (left < right) {
        int tmp    = arr[left];
        arr[left]  = arr[right];
        arr[right] = tmp;
        left++; right--;
    }
}

// 向右旋轉 k 位：三次反轉法（O(n)，無需額外陣列）
void rotateRight(int arr[], int n, int k) {
    k = k % n;             // 旋轉量不超過陣列大小
    reverse(arr, 0, n-1);  // 整體反轉
    reverse(arr, 0, k-1);  // 反轉前 k 個
    reverse(arr, k, n-1);  // 反轉後 n-k 個
}

int main() {
    int n;
    cout << "個數："; cin >> n;

    int arr[20];
    cout << "陣列：";
    for (int i = 0; i < n; i++) cin >> arr[i];

    int k;
    cout << "旋轉："; cin >> k;

    rotateRight(arr, n, k);

    cout << "旋轉後：";
    for (int i = 0; i < n; i++) {
        cout << arr[i];
        if (i < n-1) cout << " ";
    }
    cout << endl;

    return 0;
}
```

## 練習 7-3 解答：成績統計

📄 **解答檔案：** `hw7_3_grade_stats.cpp`

```cpp
#include <iostream>
#include <iomanip>
using namespace std;

int main() {
    int n;
    cout << "學生人數："; cin >> n;

    int scores[100];
    cout << "輸入成績：";
    for (int i = 0; i < n; i++) cin >> scores[i];

    int maxS = scores[0], minS = scores[0];
    double sum = 0;
    int grade[5] = {};   // F D C B A

    for (int i = 0; i < n; i++) {
        if (scores[i] > maxS) maxS = scores[i];
        if (scores[i] < minS) minS = scores[i];
        sum += scores[i];

        if      (scores[i] >= 90) grade[4]++;
        else if (scores[i] >= 80) grade[3]++;
        else if (scores[i] >= 70) grade[2]++;
        else if (scores[i] >= 60) grade[1]++;
        else                      grade[0]++;
    }

    cout << fixed << setprecision(2);
    cout << "\n最高分：" << maxS << endl;
    cout << "最低分：" << minS << endl;
    cout << "平均分：" << sum / n << endl;
    cout << "\n等級分布：" << endl;
    char labels[] = {'F','D','C','B','A'};
    for (int i = 4; i >= 0; i--) {
        cout << labels[i] << ": " << grade[i] << " 人" << endl;
    }

    return 0;
}
```

## 練習 7-4 解答：去除重複

📄 **解答檔案：** `hw7_4_remove_duplicate.cpp`

```cpp
#include <iostream>
using namespace std;

int main() {
    int n;
    cout << "個數："; cin >> n;

    int arr[100];
    cout << "陣列：";
    for (int i = 0; i < n; i++) cin >> arr[i];

    int result[100];
    int cnt = 0;

    for (int i = 0; i < n; i++) {
        bool found = false;
        for (int j = 0; j < cnt; j++) {
            if (result[j] == arr[i]) { found = true; break; }
        }
        if (!found) result[cnt++] = arr[i];
    }

    cout << "去除重複：";
    for (int i = 0; i < cnt; i++) {
        cout << result[i];
        if (i < cnt-1) cout << " ";
    }
    cout << endl;

    return 0;
}
```

## 練習 7-5 解答：二分搜尋實作

📄 **解答檔案：** `hw7_5_binary_search.cpp`

```cpp
#include <iostream>
using namespace std;

void bubbleSort(int arr[], int n) {
    for (int i = 0; i < n-1; i++) {
        bool swapped = false;
        for (int j = 0; j < n-1-i; j++) {
            if (arr[j] > arr[j+1]) {
                int t=arr[j]; arr[j]=arr[j+1]; arr[j+1]=t;
                swapped = true;
            }
        }
        if (!swapped) break;
    }
}

int binarySearch(const int arr[], int n, int target) {
    int left=0, right=n-1;
    while (left <= right) {
        int mid = left + (right-left)/2;
        if      (arr[mid] == target) return mid;
        else if (arr[mid] <  target) left  = mid+1;
        else                         right = mid-1;
    }
    return -1;
}

int main() {
    int n;
    cout << "個數："; cin >> n;
    int arr[100];
    cout << "陣列：";
    for (int i = 0; i < n; i++) cin >> arr[i];

    bubbleSort(arr, n);

    cout << "排序後：";
    for (int i = 0; i < n; i++) cout << arr[i] << " ";
    cout << endl;

    int target;
    cout << "搜尋目標："; cin >> target;

    int idx = binarySearch(arr, n, target);
    if (idx != -1)
        cout << target << " 在排序後的索引 " << idx << endl;
    else
        cout << target << " 不在陣列中" << endl;

    return 0;
}
```

## 練習 7-6 解答：矩陣旋轉 90 度

📄 **解答檔案：** `hw7_6_rotate_matrix.cpp`

```cpp
#include <iostream>
#include <iomanip>
using namespace std;

const int MAXN = 10;

int main() {
    int n;
    cout << "矩陣大小 N："; cin >> n;

    int A[MAXN][MAXN], B[MAXN][MAXN];

    cout << "輸入 " << n << "×" << n << " 矩陣：" << endl;
    for (int i = 0; i < n; i++)
        for (int j = 0; j < n; j++)
            cin >> A[i][j];

    // 順時針旋轉 90 度：B[i][j] = A[N-1-j][i]
    for (int i = 0; i < n; i++)
        for (int j = 0; j < n; j++)
            B[i][j] = A[n-1-j][i];

    cout << "\n旋轉 90 度後：" << endl;
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < n; j++) cout << setw(4) << B[i][j];
        cout << endl;
    }

    return 0;
}
```

## 練習 7-7 解答：魔術方陣驗證

📄 **解答檔案：** `hw7_7_magic_square.cpp`

```cpp
#include <iostream>
using namespace std;

int main() {
    int m[3][3];
    cout << "輸入 3×3 矩陣：" << endl;
    for (int i = 0; i < 3; i++)
        for (int j = 0; j < 3; j++)
            cin >> m[i][j];

    // 以第一列總和為基準
    int target = 0;
    for (int j = 0; j < 3; j++) target += m[0][j];

    bool magic = true;

    // 每列
    for (int i = 0; i < 3; i++) {
        int s = 0;
        for (int j = 0; j < 3; j++) s += m[i][j];
        if (s != target) { magic = false; break; }
    }

    // 每行
    if (magic) {
        for (int j = 0; j < 3; j++) {
            int s = 0;
            for (int i = 0; i < 3; i++) s += m[i][j];
            if (s != target) { magic = false; break; }
        }
    }

    // 對角線
    if (magic) {
        int d1 = m[0][0] + m[1][1] + m[2][2];
        int d2 = m[0][2] + m[1][1] + m[2][0];
        if (d1 != target || d2 != target) magic = false;
    }

    if (magic)
        cout << "是魔術方陣（總和 = " << target << "）" << endl;
    else
        cout << "不是魔術方陣" << endl;

    return 0;
}
```

## 練習 7-8 解答：螺旋矩陣

📄 **解答檔案：** `hw7_8_spiral_matrix.cpp`

```cpp
#include <iostream>
#include <iomanip>
using namespace std;

int main() {
    int n;
    cout << "N："; cin >> n;

    int m[10][10] = {};

    // 四個方向：右、下、左、上
    int dr[] = {0,  1,  0, -1};
    int dc[] = {1,  0, -1,  0};

    int r = 0, c = 0, dir = 0;

    for (int num = 1; num <= n * n; num++) {
        m[r][c] = num;

        // 試著往當前方向走一步
        int nr = r + dr[dir];
        int nc = c + dc[dir];

        // 若出界或已填入，換方向
        if (nr < 0 || nr >= n || nc < 0 || nc >= n || m[nr][nc] != 0) {
            dir = (dir + 1) % 4;
            nr = r + dr[dir];
            nc = c + dc[dir];
        }

        r = nr;
        c = nc;
    }

    for (int i = 0; i < n; i++) {
        for (int j = 0; j < n; j++) cout << setw(4) << m[i][j];
        cout << endl;
    }

    return 0;
}
```
