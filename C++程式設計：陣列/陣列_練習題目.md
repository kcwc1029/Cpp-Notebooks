### 練習：陣列反轉輸出

**題目：** 輸入 n 個整數（n ≤ 20），以相反順序輸出。

**輸入範例：**

```
請輸入個數：5
請輸入 5 個整數：3 1 4 1 5
```

**輸出範例：**

```
反轉後：5 1 4 1 3
```

<!--
```cpp
#include <iostream>
using namespace std;

int main() {
    int n;
    cout << "請輸入個數：";
    cin >> n;

    int arr[20];
    cout << "請輸入 " << n << " 個整數：";
    for (int i = 0; i < n; i++) cin >> arr[i];

    cout << "反轉後：";
    for (int i = n - 1; i >= 0; i--) {
        cout << arr[i];
        if (i > 0) cout << " ";
    }
    cout << endl;

    return 0;
}
```

### 練習 ：陣列旋轉

**題目：** 輸入 n 個整數，將陣列向右旋轉 k 個位置。

**輸入範例：**

```
個數：5
陣列：1 2 3 4 5
旋轉：2
```

**輸出範例：**

```
旋轉後：4 5 1 2 3
```

<!--
```cpp
#include <iostream>
using namespace std;

// 反轉 arr[left..right]
void reverse(int arr[], int left, int right) {
    while (left < right) {
        int tmp    = arr[left];
        arr[left]  = arr[right];
        arr[right] = tmp;
        left++; right--;
    }
}

// 向右旋轉 k 位：三次反轉法（O(n)，無需額外陣列）
void rotateRight(int arr[], int n, int k) {
    k = k % n;             // 旋轉量不超過陣列大小
    reverse(arr, 0, n-1);  // 整體反轉
    reverse(arr, 0, k-1);  // 反轉前 k 個
    reverse(arr, k, n-1);  // 反轉後 n-k 個
}

int main() {
    int n;
    cout << "個數："; cin >> n;

    int arr[20];
    cout << "陣列：";
    for (int i = 0; i < n; i++) cin >> arr[i];

    int k;
    cout << "旋轉："; cin >> k;

    rotateRight(arr, n, k);

    cout << "旋轉後：";
    for (int i = 0; i < n; i++) {
        cout << arr[i];
        if (i < n-1) cout << " ";
    }
    cout << endl;

    return 0;
}
```

**題目：** 輸入 n 個學生的成績（0~100），輸出最高分、最低分、平均分（保留 2 位小數），並統計各等級人數（A/B/C/D/F）。

<!--
```cpp
#include <iostream>
#include <iomanip>
using namespace std;

int main() {
    int n;
    cout << "學生人數："; cin >> n;

    int scores[100];
    cout << "輸入成績：";
    for (int i = 0; i < n; i++) cin >> scores[i];
```

### 練習：去除重複

**題目：** 輸入 n 個整數，輸出去除重複後的陣列（保留第一次出現的順序）。

**輸入範例：**

```
個數：8
陣列：3 1 4 1 5 9 2 6
```

**輸出範例：**（只輸出不重複的元素，保留第一次出現順序）

```
去除重複：3 1 4 5 9 2 6
```

<!--
```cpp
#include <iostream>
using namespace std;

int main() {
    int n;
    cout << "個數："; cin >> n;

    int arr[100];
    cout << "陣列：";
    for (int i = 0; i < n; i++) cin >> arr[i];

    int result[100];
    int cnt = 0;

    for (int i = 0; i < n; i++) {
        bool found = false;
        for (int j = 0; j < cnt; j++) {
            if (result[j] == arr[i]) { found = true; break; }
        }
        if (!found) result[cnt++] = arr[i];
    }

    cout << "去除重複：";
    for (int i = 0; i < cnt; i++) {
        cout << result[i];
        if (i < cnt-1) cout << " ";
    }
    cout << endl;

    return 0;
}
```

### 練習 7-5：二分搜尋實作

**題目：** 實作一個完整的二分搜尋程式。先輸入 n 個整數，程式自動排序後，輸入目標值，輸出找到的位置（如有多個，回傳第一個）。

📄 **練習檔案：** `hw7_5_binary_search.cpp`

<!--
```cpp
#include <iostream>
using namespace std;

void bubbleSort(int arr[], int n) {
    for (int i = 0; i < n-1; i++) {
        bool swapped = false;
        for (int j = 0; j < n-1-i; j++) {
            if (arr[j] > arr[j+1]) {
                int t=arr[j]; arr[j]=arr[j+1]; arr[j+1]=t;
                swapped = true;
            }
        }
        if (!swapped) break;
    }
}

int binarySearch(const int arr[], int n, int target) {
    int left=0, right=n-1;
    while (left <= right) {
        int mid = left + (right-left)/2;
        if      (arr[mid] == target) return mid;
        else if (arr[mid] <  target) left  = mid+1;
        else                         right = mid-1;
    }
    return -1;
}

int main() {
    int n;
    cout << "個數："; cin >> n;
    int arr[100];
    cout << "陣列：";
    for (int i = 0; i < n; i++) cin >> arr[i];

    bubbleSort(arr, n);

    cout << "排序後：";
    for (int i = 0; i < n; i++) cout << arr[i] << " ";
    cout << endl;

    int target;
    cout << "搜尋目標："; cin >> target;

    int idx = binarySearch(arr, n, target);
    if (idx != -1)
        cout << target << " 在排序後的索引 " << idx << endl;
    else
        cout << target << " 不在陣列中" << endl;

    return 0;
}
```

### 練習 7-6：矩陣旋轉 90 度

**題目：** 輸入一個 N×N 方陣，輸出順時針旋轉 90 度後的矩陣。

**規律：** 旋轉後 B[i][j] = A[N-1-j][i]

**輸入範例（N=3）：**

```
1 2 3
4 5 6
7 8 9
```

**輸出範例：**

```
7 4 1
8 5 2
9 6 3
```

<!--
```cpp
#include <iostream>
#include <iomanip>
using namespace std;

const int MAXN = 10;

int main() {
    int n;
    cout << "矩陣大小 N："; cin >> n;

    int A[MAXN][MAXN], B[MAXN][MAXN];

    cout << "輸入 " << n << "×" << n << " 矩陣：" << endl;
    for (int i = 0; i < n; i++)
        for (int j = 0; j < n; j++)
            cin >> A[i][j];

    // 順時針旋轉 90 度：B[i][j] = A[N-1-j][i]
    for (int i = 0; i < n; i++)
        for (int j = 0; j < n; j++)
            B[i][j] = A[n-1-j][i];

    cout << "\n旋轉 90 度後：" << endl;
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < n; j++) cout << setw(4) << B[i][j];
        cout << endl;
    }

    return 0;
}
```

### 練習：魔術方陣驗證

**題目：** 輸入一個 3×3 矩陣，判斷是否為魔術方陣（每行、每列、兩條對角線的總和都相等）。

**輸入範例：**

```
2 7 6
9 5 1
4 3 8
```

**輸出範例：**

```
是魔術方陣（總和 = 15）
```

<!--
```cpp
#include <iostream>
using namespace std;

int main() {
    int m[3][3];
    cout << "輸入 3×3 矩陣：" << endl;
    for (int i = 0; i < 3; i++)
        for (int j = 0; j < 3; j++)
            cin >> m[i][j];

    // 以第一列總和為基準
    int target = 0;
    for (int j = 0; j < 3; j++) target += m[0][j];

    bool magic = true;

    // 每列
    for (int i = 0; i < 3; i++) {
        int s = 0;
        for (int j = 0; j < 3; j++) s += m[i][j];
        if (s != target) { magic = false; break; }
    }

    // 每行
    if (magic) {
        for (int j = 0; j < 3; j++) {
            int s = 0;
            for (int i = 0; i < 3; i++) s += m[i][j];
            if (s != target) { magic = false; break; }
        }
    }

    // 對角線
    if (magic) {
        int d1 = m[0][0] + m[1][1] + m[2][2];
        int d2 = m[0][2] + m[1][1] + m[2][0];
        if (d1 != target || d2 != target) magic = false;
    }

    if (magic)
        cout << "是魔術方陣（總和 = " << target << "）" << endl;
    else
        cout << "不是魔術方陣" << endl;

    return 0;
}
```

### 練習 7-8：螺旋矩陣

**題目：** 輸入 N，生成 N×N 的螺旋矩陣（從 1 到 N² 依螺旋方向填入）。

**輸入範例：** `N = 4`

**輸出範例：**

```
 1  2  3  4
12 13 14  5
11 16 15  6
10  9  8  7
```

<!--
```cpp
#include <iostream>
#include <iomanip>
using namespace std;

int main() {
    int n;
    cout << "N："; cin >> n;

    int m[10][10] = {};

    // 四個方向：右、下、左、上
    int dr[] = {0,  1,  0, -1};
    int dc[] = {1,  0, -1,  0};

    int r = 0, c = 0, dir = 0;

    for (int num = 1; num <= n * n; num++) {
        m[r][c] = num;

        // 試著往當前方向走一步
        int nr = r + dr[dir];
        int nc = c + dc[dir];

        // 若出界或已填入，換方向
        if (nr < 0 || nr >= n || nc < 0 || nc >= n || m[nr][nc] != 0) {
            dir = (dir + 1) % 4;
            nr = r + dr[dir];
            nc = c + dc[dir];
        }

        r = nr;
        c = nc;
    }

    for (int i = 0; i < n; i++) {
        for (int j = 0; j < n; j++) cout << setw(4) << m[i][j];
        cout << endl;
    }

    return 0;
}
```
