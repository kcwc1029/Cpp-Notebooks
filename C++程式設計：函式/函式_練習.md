### 練習：溫度轉換函式

撰寫三個函式：

- `double celsiusToFahrenheit(double c)` 攝氏轉華氏，`F = C * 9 / 5 + 32`
- `double fahrenheitToCelsius(double f)` 華氏轉攝氏
- `double celsiusToKelvin(double c)` 攝氏轉克氏，`K = C + 273.15`

主程式請使用者輸入攝氏溫度，輸出三種計算結果。

#### 參考答案

```cpp
#include <iostream>
using namespace std;

double celsiusToFahrenheit(double c) {
    return c * 9.0 / 5.0 + 32.0;
}

double fahrenheitToCelsius(double f) {
    return (f - 32.0) * 5.0 / 9.0;
}

double celsiusToKelvin(double c) {
    return c + 273.15;
}

int main() {
    double c;

    cout << "請輸入攝氏溫度：";
    cin >> c;

    double f = celsiusToFahrenheit(c);

    cout << "攝氏：" << c << endl;
    cout << "華氏：" << f << endl;
    cout << "克氏：" << celsiusToKelvin(c) << endl;
    cout << "華氏再轉回攝氏：" << fahrenheitToCelsius(f) << endl;

    return 0;
}
```

### 練習：字串處理函式

撰寫以下函式（不得使用 `<algorithm>` 或 `<string>` 的現成函式，只能用索引操作）：

- `int strLen(const char* s)` 計算 C 字串長度
- `bool isPalindrome(const string& s)` 判斷迴文（忽略大小寫）
- `int countChar(const string& s, char c)` 計算字元出現次數

#### 參考答案

```cpp
#include <iostream>
#include <string>
using namespace std;

char toLowerChar(char ch) {
    if (ch >= 'A' && ch <= 'Z') {
        return ch + ('a' - 'A');
    }
    return ch;
}

int strLen(const char* s) {
    int len = 0;
    while (s[len] != '\0') {
        len++;
    }
    return len;
}

bool isPalindrome(const string& s) {
    int left = 0;
    int right = 0;

    while (s[right] != '\0') {
        right++;
    }
    right--;

    while (left < right) {
        if (toLowerChar(s[left]) != toLowerChar(s[right])) {
            return false;
        }
        left++;
        right--;
    }

    return true;
}

int countChar(const string& s, char c) {
    int count = 0;

    for (int i = 0; s[i] != '\0'; i++) {
        if (s[i] == c) {
            count++;
        }
    }

    return count;
}

int main() {
    string s;
    char c;

    cout << "請輸入字串：";
    getline(cin, s);

    cout << "請輸入要計算的字元：";
    cin >> c;

    cout << "字串長度：" << strLen(s.c_str()) << endl;
    cout << "是否為迴文：" << (isPalindrome(s) ? "是" : "否") << endl;
    cout << c << " 出現次數：" << countChar(s, c) << endl;

    return 0;
}
```

### 練習：數學函式庫

實作以下數學函式（不使用 `<cmath>`）：

- `long long power(long long base, int exp)` 整數次方（`exp >= 0`）
- `double mysqrt(double n)` 牛頓法求平方根，誤差 `< 1e-9`
- `int gcd(int a, int b)` 輾轉相除法求最大公因數
- `int lcm(int a, int b)` 最小公倍數（利用 `gcd`）

#### 參考答案

```cpp
#include <iostream>
using namespace std;

double absolute(double x) {
    if (x < 0) {
        return -x;
    }
    return x;
}

int absoluteInt(int x) {
    if (x < 0) {
        return -x;
    }
    return x;
}

long long power(long long base, int exp) {
    long long result = 1;

    for (int i = 0; i < exp; i++) {
        result *= base;
    }

    return result;
}

double mysqrt(double n) {
    if (n < 0) {
        return -1;
    }
    if (n == 0) {
        return 0;
    }

    double x = n;
    double next = 0.5 * (x + n / x);

    while (absolute(next - x) >= 1e-9) {
        x = next;
        next = 0.5 * (x + n / x);
    }

    return next;
}

int gcd(int a, int b) {
    a = absoluteInt(a);
    b = absoluteInt(b);

    while (b != 0) {
        int r = a % b;
        a = b;
        b = r;
    }

    return a;
}

int lcm(int a, int b) {
    if (a == 0 || b == 0) {
        return 0;
    }
    return absoluteInt(a / gcd(a, b) * b);
}

int main() {
    long long base;
    int exp;
    double n;
    int a, b;

    cout << "請輸入 base 和 exp：";
    cin >> base >> exp;

    cout << "請輸入要求平方根的數：";
    cin >> n;

    cout << "請輸入兩個整數：";
    cin >> a >> b;

    cout << "power：" << power(base, exp) << endl;
    cout << "sqrt：" << mysqrt(n) << endl;
    cout << "gcd：" << gcd(a, b) << endl;
    cout << "lcm：" << lcm(a, b) << endl;

    return 0;
}
```

### 練習：陣列操作函式

實作：

- `void bubbleSort(int arr[], int n)` 冒泡排序（由小到大）
- `int binarySearch(const int arr[], int n, int target)` 二分搜尋（傳回索引，找不到傳回 `-1`）
- `double standardDeviation(const int arr[], int n)` 標準差

在這些函式實作：輸入 `n` 個整數，排序，做二分搜尋目標分數，計算標準差。

#### 參考答案

```cpp
#include <iostream>
using namespace std;

double absolute(double x) {
    if (x < 0) {
        return -x;
    }
    return x;
}

double mysqrt(double n) {
    if (n < 0) {
        return -1;
    }
    if (n == 0) {
        return 0;
    }

    double x = n;
    double next = 0.5 * (x + n / x);

    while (absolute(next - x) >= 1e-9) {
        x = next;
        next = 0.5 * (x + n / x);
    }

    return next;
}

void bubbleSort(int arr[], int n) {
    for (int i = 0; i < n - 1; i++) {
        for (int j = 0; j < n - 1 - i; j++) {
            if (arr[j] > arr[j + 1]) {
                int temp = arr[j];
                arr[j] = arr[j + 1];
                arr[j + 1] = temp;
            }
        }
    }
}

int binarySearch(const int arr[], int n, int target) {
    int left = 0;
    int right = n - 1;

    while (left <= right) {
        int mid = left + (right - left) / 2;

        if (arr[mid] == target) {
            return mid;
        }
        if (arr[mid] < target) {
            left = mid + 1;
        } else {
            right = mid - 1;
        }
    }

    return -1;
}

double standardDeviation(const int arr[], int n) {
    if (n <= 0) {
        return 0;
    }

    double sum = 0;
    for (int i = 0; i < n; i++) {
        sum += arr[i];
    }

    double mean = sum / n;
    double variance = 0;

    for (int i = 0; i < n; i++) {
        double diff = arr[i] - mean;
        variance += diff * diff;
    }

    variance /= n;
    return mysqrt(variance);
}

int main() {
    int n;
    int target;

    cout << "請輸入資料筆數：";
    cin >> n;

    int arr[100];

    cout << "請輸入 " << n << " 個整數：";
    for (int i = 0; i < n; i++) {
        cin >> arr[i];
    }

    cout << "請輸入要搜尋的目標：";
    cin >> target;

    bubbleSort(arr, n);

    cout << "排序後：";
    for (int i = 0; i < n; i++) {
        cout << arr[i] << " ";
    }
    cout << endl;

    int index = binarySearch(arr, n, target);
    cout << "搜尋結果索引：" << index << endl;
    cout << "標準差：" << standardDeviation(arr, n) << endl;

    return 0;
}
```

### 練習：傳參考練習

實作以下函式，並由輸出結果驗證透過參考傳遞：

- `void swap3(int& a, int& b, int& c)` 將 `a, b, c` 循環交換（`a` 得到 `b`，`b` 得到 `c`，`c` 得到 `a`）
- `void splitNum(double x, int& intPart, double& fracPart)` 分離整數與小數部分
- `bool quadratic(double a, double b, double c, double& x1, double& x2)` 求一元二次方程式的根（有實數根傳回 `true`，無實數根傳回 `false`）

#### 參考答案

```cpp
#include <iostream>
using namespace std;

double absolute(double x) {
    if (x < 0) {
        return -x;
    }
    return x;
}

double mysqrt(double n) {
    if (n < 0) {
        return -1;
    }
    if (n == 0) {
        return 0;
    }

    double x = n;
    double next = 0.5 * (x + n / x);

    while (absolute(next - x) >= 1e-9) {
        x = next;
        next = 0.5 * (x + n / x);
    }

    return next;
}

void swap3(int& a, int& b, int& c) {
    int temp = a;
    a = b;
    b = c;
    c = temp;
}

void splitNum(double x, int& intPart, double& fracPart) {
    intPart = static_cast<int>(x);
    fracPart = x - intPart;

    if (fracPart < 0) {
        fracPart = -fracPart;
    }
}

bool quadratic(double a, double b, double c, double& x1, double& x2) {
    if (a == 0) {
        if (b == 0) {
            return false;
        }
        x1 = -c / b;
        x2 = x1;
        return true;
    }

    double discriminant = b * b - 4 * a * c;

    if (discriminant < 0) {
        return false;
    }

    double root = mysqrt(discriminant);
    x1 = (-b + root) / (2 * a);
    x2 = (-b - root) / (2 * a);

    return true;
}

int main() {
    int a, b, c;
    double x;
    int intPart;
    double fracPart;
    double qa, qb, qc;
    double x1, x2;

    cout << "請輸入三個整數：";
    cin >> a >> b >> c;

    swap3(a, b, c);
    cout << "循環交換後：" << a << " " << b << " " << c << endl;

    cout << "請輸入一個小數：";
    cin >> x;

    splitNum(x, intPart, fracPart);
    cout << "整數部分：" << intPart << endl;
    cout << "小數部分：" << fracPart << endl;

    cout << "請輸入一元二次方程式係數 a b c：";
    cin >> qa >> qb >> qc;

    if (quadratic(qa, qb, qc, x1, x2)) {
        cout << "x1 = " << x1 << endl;
        cout << "x2 = " << x2 << endl;
    } else {
        cout << "沒有實數根" << endl;
    }

    return 0;
}
```
