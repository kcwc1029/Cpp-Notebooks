## 練習 6-1 解答：基本輸出格式

📄 **解答檔案：** `hw6_1_grade_report.cpp`

```cpp
#include <iostream>
#include <iomanip>
#include <string>
using namespace std;

int main() {
    string name;
    int studentId, chinese, math, english;

    cout << "姓名：";
    cin.ignore();
    getline(cin, name);

    cout << "學號：";
    cin >> studentId;

    cout << "國文：";
    cin >> chinese;

    cout << "數學：";
    cin >> math;

    cout << "英文：";
    cin >> english;

    double avg = (chinese + math + english) / 3.0;

    cout << "╔══════════════════════════════╗" << endl;
    cout << "║          學生成績單           ║" << endl;
    cout << "╠══════════════════════════════╣" << endl;
    cout << "║ 姓名：" << left << setw(23) << name << "║" << endl;
    cout << "║ 學號：" << left << setw(23) << studentId << "║" << endl;
    cout << "╠══════════════════════════════╣" << endl;
    cout << "║ 國文：" << right << setw(4) << chinese << " 分"
         << setw(17) << left << "" << "║" << endl;
    cout << "║ 數學：" << right << setw(4) << math    << " 分"
         << setw(17) << left << "" << "║" << endl;
    cout << "║ 英文：" << right << setw(4) << english << " 分"
         << setw(17) << left << "" << "║" << endl;
    cout << "╠══════════════════════════════╣" << endl;
    cout << fixed << setprecision(2);
    cout << "║ 平均：" << right << setw(7) << avg << " 分"
         << setw(14) << left << "" << "║" << endl;
    cout << "╚══════════════════════════════╝" << endl;

    return 0;
}
```

## 練習 6-3 解答：BMI 計算機

📄 **解答檔案：** `hw6_3_bmi.cpp`

```cpp
#include <iostream>
#include <iomanip>
#include <string>
using namespace std;

int main() {
    double weight;
    int height;

    cout << "請輸入體重（kg）：";
    cin >> weight;
    cout << "請輸入身高（cm）：";
    cin >> height;

    double heightM = height / 100.0;
    double bmi = weight / (heightM * heightM);

    string level;
    if      (bmi < 18.5) level = "過輕";
    else if (bmi < 25.0) level = "正常";
    else if (bmi < 30.0) level = "過重";
    else                 level = "肥胖";

    cout << fixed << setprecision(1);
    cout << "體重：" << weight << " kg" << endl;

    cout << defaultfloat;
    cout << "身高：" << height << " cm" << endl;

    cout << fixed << setprecision(2);
    cout << "BMI：" << bmi << endl;
    cout << "體重等級：" << level << endl;

    return 0;
}
```

## 練習 6-4 解答：RGB 轉十六進位

📄 **解答檔案：** `hw6_4_rgb_hex.cpp`

```cpp
#include <iostream>
#include <iomanip>
using namespace std;

int main() {
    int r, g, b;

    cout << "R：";  cin >> r;
    cout << "G：";  cin >> g;
    cout << "B：";  cin >> b;

    // HTML 顏色代碼：#RRGGBB（大寫）
    cout << "HTML 顏色代碼：#"
         << uppercase << hex << setfill('0')
         << setw(2) << r
         << setw(2) << g
         << setw(2) << b
         << endl;

    // 帶前綴的十六進位（小寫）
    cout << nouppercase << showbase;
    cout << "十六進位（帶前綴）：R=" << hex << r
         << ", G=" << g
         << ", B=" << b << endl;

    return 0;
}
```

## 練習 6-5 解答：乘法表格

📄 **解答檔案：** `hw6_5_times_table.cpp`

```cpp
#include <iostream>
#include <iomanip>
using namespace std;

int main() {
    int n;

    cout << "請輸入 n（1~9）：";
    cin >> n;

    cout << n << " 的乘法表" << endl;
    cout << string(18, '\xe2\x94\x80') << endl;   // ──────────────────

    for (int i = 1; i <= 9; i++) {
        cout << n << " × " << i << " = "
             << setw(2) << n * i << endl;
    }

    return 0;
}
```

> **注意：** 中文破折線 `─` 在某些系統可能顯示不同，可直接用 ASCII 的 `-` 替代：
> ```cpp
> cout << string(18, '-') << endl;
> ```

## 練習 6-6 解答：個人資料卡

📄 **解答檔案：** `hw6_6_profile_card.cpp`

```cpp
#include <iostream>
#include <iomanip>
#include <string>
using namespace std;

int main() {
    string name, department, phone;
    int age;
    double gpa;

    cout << "姓名：";
    getline(cin, name);

    cout << "年齡：";
    cin >> age;

    cin.ignore();

    cout << "科系：";
    getline(cin, department);

    cout << "GPA：";
    cin >> gpa;

    cin.ignore();

    cout << "電話：";
    getline(cin, phone);

    // 固定欄寬為 31 個字元（含兩端的 │）
    cout << "┌─────────────────────────────┐" << endl;
    cout << "│        個人資料卡            │" << endl;
    cout << "├─────────────────────────────┤" << endl;

    auto row = [](string label, string value) {
        string line = "│ " + label + value;
        // 補空格對齊至 31 個字元（含 │）
        int pad = 31 - (int)line.size() - 1;
        if (pad < 0) pad = 0;
        cout << line << string(pad, ' ') << "│" << endl;
    };

    row("姓名：", name);
    row("年齡：", to_string(age) + " 歲");
    row("科系：", department);
    row("GPA： ", to_string(gpa).substr(0, 4));
    row("電話：", phone);

    cout << "└─────────────────────────────┘" << endl;

    return 0;
}
```

## 練習 6-7 解答：貨幣換算表

📄 **解答檔案：** `hw6_7_currency.cpp`

```cpp
#include <iostream>
#include <iomanip>
#include <string>
using namespace std;

int main() {
    int twd;
    cout << "請輸入台幣金額：";
    cin >> twd;

    // 匯率
    double usd = twd / 30.0;
    double jpy = twd * 4.5;
    double eur = twd / 33.0;
    double gbp = twd / 38.0;

    cout << "台幣：" << twd << " 元" << endl;
    cout << string(27, '=') << endl;
    cout << left << setw(10) << "幣別"
         << right << setw(14) << "換算金額" << endl;
    cout << string(27, '-') << endl;

    cout << fixed << setprecision(2);
    cout << left << setw(10) << "USD" << right << setw(14) << usd << endl;
    cout << left << setw(10) << "JPY" << right << setw(14) << jpy << endl;
    cout << left << setw(10) << "EUR" << right << setw(14) << eur << endl;
    cout << left << setw(10) << "GBP" << right << setw(14) << gbp << endl;
    cout << string(27, '=') << endl;

    return 0;
}
```

## 練習 6-8 解答：輸入驗證器

📄 **解答檔案：** `hw6_8_input_validator.cpp`

```cpp
#include <iostream>
using namespace std;

int main() {
    int num;
    bool valid = false;

    while (!valid) {
        cout << "請輸入 1 到 100 之間的整數：";
        cin >> num;

        if (cin.fail()) {
            // 輸入的不是整數
            cerr << "[錯誤] 請輸入整數！" << endl;
            cin.clear();              // 清除錯誤旗標
            cin.ignore(1000, '\n');   // 丟棄錯誤輸入
        } else if (num < 1 || num > 100) {
            // 整數但超出範圍
            cerr << "[錯誤] 數值超出範圍（1~100）！" << endl;
            cin.ignore(1000, '\n');
        } else {
            valid = true;
        }
    }

    cout << "輸入成功：" << num << endl;

    return 0;
}
```
