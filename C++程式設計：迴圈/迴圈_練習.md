### 輸入一個正整數 n，判斷它是否為完全平方數（例如 16 = 4²，是完全平方數）。

<!--
```cpp
#include <iostream>
#include <cmath>
using namespace std;

    int n;
    cout << "請輸入正整數：";
    cin >> n;

    int root = (int)sqrt((double)n);
    if (root * root == n) {
        cout << n << " 是完全平方數（" << root << "²）" << endl;
    } else {
        cout << n << " 不是完全平方數" << endl;
```

### 印出 1 到 100 中所有的質數，每行印 10 個，並在最後顯示共有幾個質數。

<!--
```cpp
#include <iostream>
using namespace std;

int main() {
    int count = 0;

    for (int n = 2; n <= 100; n++) {
        bool isPrime = true;
        for (int i = 2; i * i <= n; i++) {
            if (n % i == 0) {
                isPrime = false;
                break;
            }
        }
        if (isPrime) {
            cout << n;
            count++;
            if (count % 10 == 0) {
                cout << endl;
            } else {
                cout << "\t";
            }
        }
    }

    cout << endl << "共 " << count << " 個質數" << endl;
    return 0;
}
```

### 用巢狀迴圈印出以下菱形（n 由使用者輸入，下面為 n=4 的範例）：

```
   *
  ***
 *****
*******
 *****
  ***
   *
```

<!--
```cpp
#include <iostream>
using namespace std;

int main() {
    int n;
    cout << "請輸入 n：";
    cin >> n;

    // 上半部（含中間行）
    for (int i = 1; i <= n; i++) {
        for (int j = 1; j <= n - i; j++) cout << " ";
        for (int j = 1; j <= 2 * i - 1; j++) cout << "*";
        cout << endl;
    }

    // 下半部
    for (int i = n - 1; i >= 1; i--) {
        for (int j = 1; j <= n - i; j++) cout << " ";
        for (int j = 1; j <= 2 * i - 1; j++) cout << "*";
        cout << endl;
    }

    return 0;
}
```

### 找出 1000 以內所有的**完美數**（perfect number）。完美數的定義：一個數等於其所有真因數（不含本身）的總和。例如：6 = 1+2+3，28 = 1+2+4+7+14。

<!--
```cpp
#include <iostream>
using namespace std;

int main() {
    cout << "1000 以內的完美數：" << endl;

    for (int n = 2; n <= 1000; n++) {
        int sum = 1;    // 1 必定是因數

        for (int i = 2; i * i <= n; i++) {
            if (n % i == 0) {
                sum += i;
                if (i != n / i) {    // 避免重複加平方根
                    sum += n / i;
                }
            }
        }

        if (sum == n) {
            cout << n << " 是完美數（";

            // 列出因數
            bool first = true;
            for (int i = 1; i < n; i++) {
                if (n % i == 0) {
                    if (!first) cout << "+";
                    cout << i;
                    first = false;
                }
            }
            cout << "=" << n << "）" << endl;
        }
    }

    return 0;
}
```

### 印出 n×n 的空心正方形，例如 n=5：

```
*****
*   *
*   *
*   *
*****
```

<!--
```cpp
#include <iostream>
using namespace std;

int main() {
    int n;
    cout << "請輸入 n：";
    cin >> n;

    for (int i = 0; i < n; i++) {
        for (int j = 0; j < n; j++) {
            // 邊框：第一行、最後行、第一列、最後列
            if (i == 0 || i == n - 1 || j == 0 || j == n - 1) {
                cout << "*";
            } else {
                cout << " ";
            }
        }
        cout << endl;
    }

    return 0;
}
```
